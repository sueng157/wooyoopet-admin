function create_room(object $request)
{
    $members = [$request->mb_id, $request->to_mb_id];
    sort($members);

    $sql = "
        SELECT mb_id, mb_5
        FROM g5_member
        WHERE mb_id = '" . $request->mb_id . "'
    ";
    $result = sql_fetch($sql);
    $memeberInfo = [
        [
            'mb_id' => $request->mb_id,
            'mb_5' => $result['mb_5'],
        ],
        [
            'mb_id' => $request->to_mb_id,
            'mb_5' => $result['mb_5'] == '1' ? '2' : '1',
        ]
    ];

    $roomName = [];
    foreach ($members as $value) {
        $roomName[] = $value . '-' . $memeberInfo[array_search($value, array_column($memeberInfo, 'mb_id'))]['mb_5'];
    }
    $roomName = implode(',', $roomName);
    
    $query = "
        SELECT *
        FROM room
        WHERE 
            EXISTS (
                SELECT 1
                FROM room_members
                WHERE room_members.room_id = room.id
                GROUP BY room_id
                HAVING GROUP_CONCAT(DISTINCT CONCAT(mb_id, '-', mb_5) ORDER BY mb_id) = '" . $roomName . "' 
            )
            AND room.deleted_at IS NULL
    ";
    $room = sql_fetch($query);

    if (!$room) {
        // 채팅방이 없으면 생성
        $query = "
            INSERT INTO room (name)
            VALUES ('" . $roomName . "')
        ";
        $result = sql_query($query);

        if ($result) {
            $roomId = sql_insert_id();

            foreach ($memeberInfo as $value) {
                $query = "
                    INSERT INTO room_members (room_id, mb_id, mb_5)
                    VALUES ('{$roomId}', '{$value['mb_id']}', '{$value['mb_5']}')
                ";
                sql_query($query);
            }

            $room = sql_fetch("SELECT * FROM room WHERE id = '{$roomId}'");
        } else {
            throw new Exception('채팅방 생성에 실패했습니다.');
        }
    }

    $room = relationMemeber($room, $room['id']);

    return ['data' => $room];
}

function get_rooms(object $request)
{
    $mb_id = $request->mb_id;
    $is_unread_filter = $request->is_unread == 'true';

    $query = "
        SELECT 
            room.*,
            (
                SELECT COUNT(*) 
                FROM chat 
                WHERE chat.room_id = room.id
                    AND chat.id > IFNULL(room_members.last_read_message_id, 0)
                    AND chat.mb_id != '{$mb_id}'
                    AND chat.mb_id != 'system'
            ) AS unread_count,
            room_members.is_muted
        FROM room
        JOIN room_members ON room_members.room_id = room.id
        JOIN g5_member ON g5_member.mb_id = room_members.mb_id
        WHERE room_members.mb_id = '{$mb_id}'
            AND room_members.mb_5 = g5_member.mb_5
            AND EXISTS (
                SELECT 1
                FROM chat
                WHERE chat.room_id = room.id
            )
            AND EXISTS (
                SELECT 1
                FROM room_members
                WHERE room_members.room_id = room.id
                    AND room_members.mb_id != '{$mb_id}'
                    AND EXISTS (
                        SELECT 1
                        FROM g5_member AS t
                        WHERE t.mb_id = room_members.mb_id
                    )
            )
            AND room.deleted_at IS NULL
    ";
    if ($is_unread_filter) {
        $query .= "
            AND (
                SELECT COUNT(*) 
                FROM chat 
                WHERE chat.room_id = room.id
                    AND chat.id > IFNULL(room_members.last_read_message_id, 0)
                    AND chat.mb_id != '{$mb_id}'
                    AND chat.mb_id != 'system'
            ) > 0
        ";
    }

    $query .= "
        ORDER BY 
            (
                SELECT MAX(created_at) 
                FROM chat 
                WHERE chat.room_id = room.id
            ) DESC
    ";

    $rooms = [];
    $result = sql_query($query);
    while ($row = sql_fetch_array($result)) {
        $query = "
            SELECT *
            FROM chat
            WHERE room_id = '{$row['id']}'
                AND mb_id != 'system'
            ORDER BY created_at DESC
            LIMIT 1
        ";
        $lastMessage = sql_fetch($query);

        $row = relationMemeber($row, $row['id']);
        $row['last_message'] = $lastMessage;

        $rooms[] = $row;
    }

    return ['data' => $rooms];
}

    $sql = "
        UPDATE room_members
        SET last_read_message_id = (
            SELECT MAX(id)
            FROM chat
            WHERE room_id = '{$roomId}'
        )
        WHERE room_id = '{$roomId}'
            AND mb_id = '{$mbId}'
    ";
    sql_query($sql);

    return [
        'data' => $messages,
        'meta' => [
            'page' => (int) $page,
            'perPage' => (int) $perPage,
            'total' => (int) $total,
        ],
    ];

    send($row['mb_id'], '돌봄이 종료되었습니다.', '거래 후기를 작성해주세요.', ['type' => 'care_completed']);
    send($row['to_mb_id'], '돌봄이 종료되었습니다.', '거래 후기를 작성해주세요.', ['type' => 'care_completed']);

    // 돌봄 종료 시 상태를 care_completed 로 변경
    sql_query("UPDATE payment_request SET status = 'care_completed', updated_at = NOW() WHERE id = '{$pid}'");
}

<?php
/**
 * 정산 정보 주민번호 뒷자리 암호화/복호화 (AES-256-CBC)
 * include 전에 common.php(config) 로드 필요.
 */

if (!function_exists('settlement_rrn_encrypt')) {
    function settlement_rrn_encrypt($plain) {
        $key = (defined('G5_SETTLEMENT_ENCRYPT_KEY') && G5_SETTLEMENT_ENCRYPT_KEY !== '')
            ? hash('sha256', G5_SETTLEMENT_ENCRYPT_KEY, true)
            : hash('sha256', (defined('G5_MYSQL_USER') ? G5_MYSQL_USER : '') . 'settlement_rrn_enc', true);
        $iv = openssl_random_pseudo_bytes(16);
        $enc = openssl_encrypt($plain, 'aes-256-cbc', $key, OPENSSL_RAW_DATA, $iv);
        if ($enc === false) return '';
        return base64_encode($iv . $enc);
    }
}

if (!function_exists('settlement_rrn_decrypt')) {
    function settlement_rrn_decrypt($cipherBase64) {
        if ($cipherBase64 === '' || $cipherBase64 === null) return '';
        $raw = base64_decode($cipherBase64, true);
        if ($raw === false || strlen($raw) < 17) return '';
        $key = (defined('G5_SETTLEMENT_ENCRYPT_KEY') && G5_SETTLEMENT_ENCRYPT_KEY !== '')
            ? hash('sha256', G5_SETTLEMENT_ENCRYPT_KEY, true)
            : hash('sha256', (defined('G5_MYSQL_USER') ? G5_MYSQL_USER : '') . 'settlement_rrn_enc', true);
        $iv = substr($raw, 0, 16);
        $cipher = substr($raw, 16);
        $dec = openssl_decrypt($cipher, 'aes-256-cbc', $key, OPENSSL_RAW_DATA, $iv);
        return $dec !== false ? $dec : '';
    }
}
